## Approach
```python
class Solution:

	def deleteNode(self, root: Optional[TreeNode], key: int) -> Optional[TreeNode]:
	
		# FIXME This is used to insert a single node
		# We need to make sure that:
		# ALL the nodes of the LHS are less
		# ALL the nodes of the RHS are greater
		def insert(root, node):
			if not root:
				return node
			
			if node.val < root.val:
				root.left = insert(root.left, node)
			
			if node.val > root.val:
				root.right = insert(root.right, node)
		  
		
		def delete(root, key):
		
			if not root:
				return root
			
			if key < root.val:
				root.left = delete(root.left, key)
			
			if key > root.val:
				root.right = delete(root.right, key)
			
			# if the value is the same as the key
			if key == root.val:
				left = root.left if root.left else None
				right = root.right if root.right else None
				
				if right and left:
					return insert(right, left)
				
				return right if right else None
				# FIXME this is unreachable because the cases are exhaustive ^^
				return left if left else None
		
			return root
	
		return delete(root, key)
```

##### Mistakes:
- **Binary tree invariant:**
	- LEFT: All the nodes are smaller than it 
	- RIGHT: All the nodes are greater than it
